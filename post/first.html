class Solution {
public:
    double angleClock(int hour, int minutes) {
        double res;
        double res_hour = hour*30+minutes/2.0;
        double res_min = minutes * 6;
        res =  max(res_hour,res_min) - min(res_hour,res_min);
        res = min(res,360-res);
        return res;
    }
};
